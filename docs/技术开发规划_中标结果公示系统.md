```python
# src/core/fetcher.py 示例
import httpx
from tenacity import retry, stop_after_attempt, wait_exponential
from typing import Self
import structlog

logger = structlog.get_logger()

class FetcherConfig(BaseModel):
    """HTTP客户端配置"""
    timeout: float = 30.0
    max_retries: int = 3
    min_retry_wait: float = 1.0
    max_retry_wait: float = 60.0
    user_agent: str = "Mozilla/5.0 (compatible; GovernmentScraper/1.0)"

class Fetcher:
    """HTTP请求封装，支持自动重试和限流"""

    def __init__(self, config: FetcherConfig) -> None:
        self._config = config
        self._client: httpx.Client | None = None

    def __enter__(self) -> Self:
        self._client = httpx.Client(
            timeout=httpx.Timeout(self._config.timeout),
            headers={"User-Agent": self._config.user_agent},
        )
        return self

    def __exit__(self, *args: object) -> None:
        self._client.close()

    @retry(
        stop=stop_after_attempt(3),
        wait=wait_exponential(multiplier=1, min=1, max=60),
        retry=retry_if_exception_type(httpx.HTTPError),
    )
    async def fetch(self, url: str) -> httpx.Response:
        """带重试的GET请求"""
        if self._client is None:
            raise RuntimeError("Fetcher not initialized. Use 'async with'.")

        logger.info("fetching_url", url=url)
        response = self._client.get(url)
        response.raise_for_status()
        return response
```

```python
# src/crawlers/base.py
from abc import ABC, abstractmethod
from typing import Generic, TypeVar
from dataclasses import dataclass

T = TypeVar("T")

@dataclass
class CrawlResult(Generic[T]):
    """爬取结果封装"""
    success: bool
    data: T | None = None
    error: str | None = None
    retry_count: int = 0

class BaseCrawler(ABC, Generic[T]):
    """采集数据基类"""

    def __init__(
        self,
        fetcher: Fetcher,
        parser: Parser,
        rate_limiter: RateLimiter,
    ) -> None:
        self._fetcher = fetcher
        self._parser = parser
        self._rate_limiter = rate_limiter

    @abstractmethod
    async def crawl_list(self, page: int, **kwargs) -> CrawlResult[list[T]]:
        """抓取列表页"""
        pass

    @abstractmethod
    async def crawl_detail(self, item_id: str) -> CrawlResult[T]:
        """抓取详情页"""
        pass

    async def crawl_with_rate_limit(self, url: str) -> str:
        """带限流的抓取"""
        await self._rate_limiter.acquire()
        response = await self._fetcher.fetch(url)
        return response.text
```

```python
from apscheduler.schedulers.asyncio import AsyncIOScheduler
from apscheduler.triggers.cron import CronTrigger

def create_scheduler() -> AsyncIOScheduler:
    scheduler = AsyncIOScheduler()

    # 每天凌晨2点执行增量抓取
    scheduler.add_job(
        incremental_crawl,
        CronTrigger(hour=2, minute=0),
        id="incremental_crawl",
        replace_existing=True,
    )

    # 每周日凌晨3点执行全量抓取
    scheduler.add_job(
        full_crawl,
        CronTrigger(day_of_week="sun", hour=3, minute=0),
        id="full_crawl",
        replace_existing=True,
    )

    return scheduler
```
